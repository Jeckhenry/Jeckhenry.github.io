js原生实现图片懒加载

发表于 2018-06-22
通过监听元素是否出现在可视区域内来加载图片，以达到优化性能的目的

    实现思路，首先将图片地址置于data-sc属性中，初始时不加载图片。添加全局监听器，当元素出现在可视区域内时，通过getAttribute、dataset.src取得图片地址，
生成img标签并追加到页面上。

创建img标签，并把data-src值置于img的src属性；
// 创建img标签
function createImg(obj){ 
    var src = '';
    if(obj.dataset.src){
        src = obj.dataset.src;
    }else{
        src = obj.getAttribute('data-src');
    }
    if(obj.children.length<=1){
        var img = document.createElement('img');
        img.src = src;
        obj.appendChild(img)
    }
}


获取元素到文档顶端的距离
//计算节点到文档顶部的距离
function getTop(obj){
    var h = 0;
    while(obj){
        h+=obj.offsetTop;
        obj = obj.offsetParent;
    }
    return h;
}

//滚动时实时计算滚动区域并进行相关运算
window.onscroll = function(){
    var t = document.documentElement.clientHeight + (document.body.scrollTop || document.documentElement.scrollTop);
    for(var i=0;i<oLI.length;i++){
        if(getTop(oLI[i]) < t){
            setTimeout('createImg(oLI['+i+'])',1000);
        }
    }
}

页面加载完成后添加滚动监听
window.onload = function(){
    window.onscroll();
}
